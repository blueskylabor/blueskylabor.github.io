&E=\frac{1}{2}(O^1_0-t)^2\

&=(O_0-t)\frac{\delta O_0}{\delta w_{j0}}\
&=(O_0-t)\delta(x_0)(1-\delta(x_0))\frac{\delta x_0^1}{\delta w_j^0}\
&=(O_0-t)O_0(1-O_0)\frac{\delta x_0^1}{\delta w_j^0}\
&=(O_0-t)O_0(1-O_0)x_j^0
\end{aligned}
$$
